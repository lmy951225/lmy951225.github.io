9void init_eCAN_B ( uint16_T bitRatePrescaler, uint16_T timeSeg1, uint16_T



10                  timeSeg2, uint16_T sbg, uint16_T sjw, uint16_T sam)



11{



12  struct ECAN_REGS ECanbShadow;



13



14  /* Configure CAN pins using GPIO regs */



15  EALLOW;



16  GpioCtrlRegs.GPAPUD.bit.GPIO17 = 0;  /* Enable pull-up for GPIO17 (CANRXB)*/



17  GpioCtrlRegs.GPAMUX2.bit.GPIO17 = 2; /* Configure GPIO17 for CANRXB operation*/



18  GpioCtrlRegs.GPAPUD.bit.GPIO16 = 0;  /* Enable pull-up for GPIO16  (CANTXB) */



19  GpioCtrlRegs.GPAMUX2.bit.GPIO16 = 2; /* Configure GPIO16 for CANTXB operation*/



20  EDIS;



21



22  /* Configure the RX and TX pins for transmission */



23  EALLOW;



24  ECanbShadow.CANRIOC.all = ECanbRegs.CANRIOC.all;



25  ECanbShadow.CANRIOC.bit.RXFUNC = 1;



26  ECanbRegs.CANRIOC.all = ECanbShadow.CANRIOC.all;



27  ECanbShadow.CANTIOC.all = ECanbRegs.CANTIOC.all;



28  ECanbShadow.CANTIOC.bit.TXFUNC = 1;



29  ECanbRegs.CANTIOC.all = ECanbShadow.CANTIOC.all;



30  EDIS;



31



32  /* Disable all Mailboxes */



33  ECanbRegs.CANME.all = 0;



34  EALLOW;



35  ECanbRegs.CANMIM.all = 0xFFFFFFFF;



36  ECanbShadow.CANMC.all = ECanbRegs.CANMC.all;



37  ECanbShadow.CANMC.bit.CCR = 1;



38  ECanbRegs.CANMC.all = ECanbShadow.CANMC.all;



39  EDIS;



40  do {



41    ECanbShadow.CANES.all = ECanbRegs.CANES.all;



42  } while (ECanbShadow.CANES.bit.CCE != 1 );/* Wait for CCE bit to be set..*/



43



44  /* Configure the eCAN timing parameters */



45  EALLOW;



46  ECanbShadow.CANBTC.all = ECanbRegs.CANBTC.all;



47  ECanbShadow.CANBTC.bit.BRPREG = bitRatePrescaler-1;



48  ECanbShadow.CANBTC.bit.TSEG2REG = timeSeg2-1;



49  ECanbShadow.CANBTC.bit.TSEG1REG = timeSeg1-1;



50  ECanbShadow.CANBTC.bit.SAM = sam-1;



51  ECanbShadow.CANBTC.bit.SJWREG = sjw-1;



52



53  /*ECanbShadow.CANBTC.bit.ERM = sbg-1; */



54  ECanbRegs.CANBTC.all = ECanbShadow.CANBTC.all;



55  ECanbShadow.CANMC.all = ECanbRegs.CANMC.all;



56  ECanbShadow.CANMC.bit.CCR = 0 ;      /* Set CCR = 0*/



57  ECanbRegs.CANMC.all = ECanbShadow.CANMC.all;



58  EDIS;



59  do {



60    ECanbShadow.CANES.all = ECanbRegs.CANES.all;



61  } while (ECanbShadow.CANES.bit.CCE != 0 );/* Wait for CCE bit to be cleared..*/



62



63  EALLOW;



64  ECanbShadow.CANMC.all = ECanbRegs.CANMC.all;



65  ECanbShadow.CANMC.bit.STM = 0;       /* Configure CAN for normal mode    */



66  ECanbShadow.CANMC.bit.SCB = 1;       /* User has selected eCAN mode*/



67  ECanbShadow.CANMC.bit.DBO = 1;       /* Set byte order: least significant first*/



68  ECanbRegs.CANMC.all = ECanbShadow.CANMC.all;



69  EDIS;



70



71  /* Configure CAN interrupts */



72  EALLOW;



73  ECanbShadow.CANMIM.all = ECanbRegs.CANMIM.all;



74  ECanbShadow.CANMIM.all = (uint32_T) 0;



75  ECanbRegs.CANMIM = ECanbShadow.CANMIM;



76  EDIS;



77  EALLOW;



78  ECanbShadow.CANMIL.all = ECanbRegs.CANMIL.all;



79  ECanbShadow.CANMIL.all = (uint32_T) 0;



80  ECanbRegs.CANMIL = ECanbShadow.CANMIL;



81  EDIS;



82  EALLOW;



83  ECanbShadow.CANGIM.all = ECanbRegs.CANGIM.all;



84  ECanbShadow.CANGIM.bit.GIL = 0;      /* GIL value determines ECanb(0/1)INT*/



85  ECanbShadow.CANGIM.bit.I0EN = 0;



86  ECanbShadow.CANGIM.bit.I1EN = 0;



87  ECanbRegs.CANGIM.all = ECanbShadow.CANGIM.all;



88  EDIS;



89}

91void config_eCAN_B_mbx (uint16_T mbxType, uint16_T mbxNo, uint32_T msgID,



92  uint16_T msgType)



93{



94  uint32_T maskRx = 0x1;



95  uint32_T maskTx;



96  struct ECAN_REGS ECanbShadow;



97  volatile struct MBOX *mbx = &ECanbMboxes.MBOX0 + mbxNo;



98  maskRx = maskRx<<mbxNo;



99  maskTx = ~maskRx;



100  mbx->MSGCTRL.bit.RTR = 0;



101  ECanbShadow.CANME.all = ECanbRegs.CANME.all;



102  ECanbRegs.CANME.all = 0x00000000;



103  if (msgType==1)



104    mbx->MSGID.all = msgID;            /* For Extended ID use bits  0:28 */



105  else



106    mbx->MSGID.bit.STDMSGID = msgID;   /* For Standard ID use bits 18:28 */



107  mbx->MSGID.bit.AME = 0;



108  mbx->MSGID.bit.AAM = 0;



109  mbx->MSGID.bit.IDE = msgType;



110  ECanbShadow.CANMD.all = ECanbRegs.CANMD.all;



111  if (mbxType==0) {



112    ECanbShadow.CANMD.all |= maskRx;



113  } else {



114    ECanbShadow.CANMD.all &= maskTx;



115  }



116



117  ECanbRegs.CANMD.all = ECanbShadow.CANMD.all;



118  ECanbShadow.CANME.all |= maskRx;



119  ECanbRegs.CANME.all = ECanbShadow.CANME.all;



120}
